def bfs(i, j, N):
    cnt = 0
    q = [(i, j)]    # 큐 생성, 시작점 인큐
    visited[i][j] = 1
    while q:
        i, j = q.pop(0)
        cnt += 1
        for di, dj in [[0,1], [1, 0], [0, -1], [-1, 0]]:
            ni, nj = i + di, j + dj
            if 0 <= ni < N and 0 <= nj < N and  arr[ni][nj] == 1 and visited[ni][nj] == 0:
                q.append((ni, nj))
                visited[ni][nj] = 1
    return cnt


N = int(input())
arr = [list(map(int, input())) for _ in range(N)]

visited = [[0] * N for _ in range(N)]
stack = []
ans = 0
cnt = 0
for i in range(N):
    for j in range(N):
        if arr[i][j] == 1 and visited[i][j] == 0:
            ans += 1
            cnt = bfs(i, j, N)
            stack.append(cnt)

print(ans)
stack.sort()
for q in stack:
    print(q)

    for j in range(N):
        if arr[i][j] == 1:
            queue = []

            cnt = 1
            arr[i][j] = 0
            while queue:
                si, sj = queue.pop(0)
                for di, dj in [[0, 1], [1, 0], [0, -1], [-1, 0]]:
                    ni, nj = si + di, sj + dj
                    if 0 <= ni < N and 0 <= nj < N and arr[ni][nj] == 1:
                        queue.append([ni, nj])
                        arr[ni][nj] = 0
                        cnt += 1
            answer.append(cnt)
answer.sort()
answer = [len(answer)] + answer
for i in answer:
